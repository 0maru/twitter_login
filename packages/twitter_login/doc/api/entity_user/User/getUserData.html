static Future<User> getUserData(
  String apiKey,
  String apiSecretKey,
  String accessToken,
  String accessTokenSecret,
) async {
  try {
    final authParams = requestHeader(
      apiKey: apiKey,
      oauthToken: accessToken,
    );
    final params = await httpGet(
      ACCOUNT_VERIFY_URI,
      authParams,
      apiKey,
      apiSecretKey,
      accessTokenSecret,
    );
    return User._(params);
  } on Exception catch (error) {
    throw Exception(error);
  }
}
